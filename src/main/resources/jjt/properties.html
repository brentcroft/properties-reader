<LF: "\n">
| <CR: "\r">
| <SPACE: (" ")+>
| <BANG: "!" (<SPACE>)*>
| <HASH: "#" (<SPACE>)*>
| <EQUALS: "=">
| <COLON: ":">
| <CONT: "\\" (~["\n","\r","\\"])*>
| <CHAR: (~["\n","\r","!","#","=",":"," "])+>
}

parse ::= ( ( ( ( eol ) | ( space ) | initiator ( text )? ( eol )? | key ( sep ( text ( continuations )? )? )? ( eol )? ) )+ | <EOF> )
initiator ::= ( <BANG> | <HASH> )
text ::= ( ( <CHAR> | <SPACE> ) )+
continuations ::= ( ( <CONT> ( eol space ( text )? )? ) )+
key ::= <CHAR>
sep ::= ( ( ( space )? <EQUALS> ( space )? ) | ( ( space )? <COLON> ( space )? ) | ( space ) )
space ::= <SPACE>
eol ::= ( <LF> | ( <CR> <LF> ) )
